    3. 然后写一个python脚本进行词云分析。参考一下。
        
        ```python
        import os
        import re
        from collections import Counter
        from wordcloud import WordCloud
        import matplotlib.pyplot as plt
        
        # 创建一个字符串，包含所有Markdown文件的内容
        text = ""
        for root, dirs, files in os.walk('.'):  # 使用os.walk递归遍历当前目录及子目录
            for filename in files:
                if filename.endswith('.md'):
                    file_path = os.path.join(root, filename)
                    with open(file_path, 'r', encoding='utf-8') as f:
                        text += f.read() + " "
                 
        # 要过滤掉的词汇列表
        filter_words = ['新增对应的任务', '当然了', '子弹']
        
        # 从文本中删除要过滤的词汇
        for word in filter_words:
            text = text.replace(word, '')
            
        # 定义一个正则表达式，匹配中文字符和常见的中文标点
        regex_pattern = r'[\u4e00-\u9fff]+'
        
        # 使用正则表达式找到所有匹配的内容
        words = re.findall(regex_pattern, text)
        
        # 统计每个词的出现次数
        word_counts = Counter(words)
        
        # 定义一个频率上限阈值
        frequency_limit = 5
        
        # 移除超过频率上限的词汇
        filtered_words = {word: count for word, count in word_counts.items() if count <= frequency_limit}
        
        # 生成词云
        wordcloud = WordCloud(font_path='/usr/share/fonts/opentype/noto/NotoSerifCJK-Regular.ttc', width=800, height=400, background_color='white').generate_from_frequencies(filtered_words)
        
        # 使用matplotlib展示词云
        plt.figure(figsize=(10, 5))
        plt.imshow(wordcloud, interpolation='bilinear')
        plt.axis('off')
        plt.savefig('wordcloud.png', dpi=1000)  # 保存词云为wordcloud.png
        plt.show()
        ```
